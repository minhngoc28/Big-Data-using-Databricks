#9 Write a function which creates an nn matrix with (i,j)-entry equal to i+j.

import numpy as np

def create_matrix(n):
    matrix = np.fromfunction(lambda i, j: i + j, (n, n), dtype=int)
    return matrix

#10 Write a function which creates an nn matrix with rows having subsequent values multiplied by the row's number. For example for n = 4:
# [[0, 1, 2, 3], [0, 2, 4, 6], [0, 3, 6, 9], [0, 4, 8, 12]]

import numpy as np

def create_matrix(n):
    matrix = np.array([[i * j for j in range(n)] for i in range(1, n + 1)])
    return matrix

# Example for n = 4
result_matrix = create_matrix(4)
print(result_matrix)
